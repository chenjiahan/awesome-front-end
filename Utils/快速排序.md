    //快速排序
    var quickSort = function(arr) {
        if(arr.length <= 1) {
            return arr;
        }
        var left  = [];
        var right = [];
        var equal = [];
        var mid   = arr[Math.floor(arr.length / 2)];
        for(var i = 0, len = arr.length; i < len; i++) {
            if(arr[i] < mid) {
                left.push(arr[i]);
            } else if(arr[i] > mid) {
                right.push(arr[i]);
            } else {
                equal.push(arr[i]);
            }
        }
        return quickSort(left).concat(equal,quickSort(right));
    }
